1.先建表

5个表

1. 首页位置表 `pre_nav`

   ```
   CREATE TABLE IF NOT EXISTS  `pre_nav`(
      `id` INT UNSIGNED AUTO_INCREMENT PRIMARY KEY COMMENT 'id',
      `position_id` TINYINT UNSIGNED NOT NULL COMMENT '位置id',
      `title`  VARCHAR(50) COMMENT '模块名',
      `picture` VARCHAR(255)  COMMENT '模块图片',
      `link_type` TINYINT UNSIGNED NOT NULL COMMENT '链接类型' ,
      `link_target` INT UNSIGNED NOT NULL  COMMENT '链接目标',
      `status` CHAR(1) NOT NULL COMMENT '状态',
      `created_at`DATETIME  COMMENT '创建时间',
      `updated_at`DATETIME  COMMENT '修改时间'
   )ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
   ```

   

2. 商品分类表`pre_category`

   ```
   CREATE TABLE IF NOT EXISTS `pre_category`(
      `id` INT UNSIGNED AUTO_INCREMENT PRIMARY KEY COMMENT '分类id' ,
      `name` VARCHAR(50)  UNIQUE   NOT NULL COMMENT '分类的名称' ,
      `property`  VARCHAR(500) NOT NULL COMMENT '属性(json数据)',
      `sort` INT NOT NULL COMMENT '排序',
       `status` CHAR(1) NOT NULL COMMENT '状态',
       `created_at`DATETIME  COMMENT '创建时间',
       `updated_at`DATETIME  COMMENT '修改时间'
   )ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
   ```

   

3. 商品库存表`pre_sku`

   ```
   CREATE TABLE IF NOT EXISTS `pre_sku`(
      `id` INT UNSIGNED AUTO_INCREMENT PRIMARY KEY COMMENT '库存id' ,
      `product_id` INT UNSIGNED COMMENT '商品id' ,
       `original_price` DECIMAL(10,2) UNSIGNED   COMMENT '原价',
       `price` DECIMAL(10,2) UNSIGNED  COMMENT '售价',
      `attr1` VARCHAR(50)  COMMENT '商品的属性1',
       `attr2`VARCHAR(50)    COMMENT '商品的属性2',
       `attr3` VARCHAR(50)   COMMENT '商品的属性3',
      `quantity` INT UNSIGNED  NOT NULL DEFAULT 0  COMMENT '商品库存',
       `sort` INT NOT NULL COMMENT '排序',
        `status` CHAR(1) NOT NULL COMMENT '状态',
       `created_at`DATETIME  COMMENT '创建时间',
       `updated_at`DATETIME  COMMENT '修改时间'
   )ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
   ```

   

4. 商品表`pre_product`

   ```
   CREATE TABLE IF NOT EXISTS `ssm_product`(
      `id` INT UNSIGNED AUTO_INCREMENT PRIMARY KEY COMMENT '商品id',
      `category_id` INT UNSIGNED NOT NULL  COMMENT '商品分类id',
      `name`  VARCHAR(50) UNIQUE NOT NULL   COMMENT '商品名称' ,
      `sale_num` INT UNSIGNED NOT NULL DEFAULT 0  COMMENT '商品的销量',
      `content` longtext   COMMENT '商品的描述',
      `sort` INT NOT NULL COMMENT '排序',
      `status` CHAR(1) NOT NULL COMMENT '状态',
      `created_at`DATETIME  COMMENT '创建时间',
      `updated_at`DATETIME  COMMENT '修改时间'
   )ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
   ```

   

5. 商品标签表`pre_product_tag`

   ```
   CREATE TABLE IF NOT EXISTS `pre_product_tag`(
      `id` INT UNSIGNED AUTO_INCREMENT PRIMARY KEY COMMENT '标签id',
      `product_id` INT UNSIGNED NOT NULL COMMENT '商品id',
      `tag_id` TINYINT UNSIGNED NOT NULL COMMENT '标签类型id',
      `value` VARCHAR(255) COMMENT '标签值',
      `status` CHAR(1) NOT NULL COMMENT '状态',
   `created_at`DATETIME  COMMENT '创建时间',
      `updated_at`DATETIME  COMMENT '修改时间'
   )ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
   ```
